  ## Usage

  Place `GradientMesh` inside a relative container. It renders overlapping blurred radial gradients that shift position via CSS animations.

  ### Basic

  <ExampleBlock
    codeDefaultOpen
    code={`import { GradientMesh } from "@glinui/ui"

export function GradientMeshDemo() {
  return (
    <div className="relative flex h-64 w-full items-center justify-center overflow-hidden rounded-xl bg-neutral-950">
      <GradientMesh />
      <span className="relative z-10 text-3xl font-bold text-white">Mesh</span>
    </div>
  )
}`}
  >
    <div className="relative flex h-64 w-full items-center justify-center overflow-hidden rounded-xl bg-neutral-950">
      <GradientMesh />
      <span className="relative z-10 text-3xl font-bold text-white">Mesh</span>
    </div>
  </ExampleBlock>
