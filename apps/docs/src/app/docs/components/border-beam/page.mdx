  ## Usage

  Wrap any container with `position: relative` and place `BorderBeam` inside. The beam travels along the border using CSS `offset-path`.

  ### Basic

  <ExampleBlock
    codeDefaultOpen
    code={`import { BorderBeam } from "@glinui/ui"

export function BorderBeamDemo() {
  return (
    <div className="relative h-40 w-full max-w-md rounded-xl border border-neutral-200 bg-white dark:border-neutral-800 dark:bg-neutral-950">
      <BorderBeam />
    </div>
  )
}`}
  >
    <div className="relative h-40 w-full max-w-md rounded-xl border border-neutral-200 bg-white dark:border-neutral-800 dark:bg-neutral-950">
      <BorderBeam />
    </div>
  </ExampleBlock>
